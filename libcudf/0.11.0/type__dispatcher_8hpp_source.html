198 template <template <cudf::type_id> typename IdTypeMap = id_to_type_impl,
199  typename Functor, typename... Ts>
200 CUDA_HOST_DEVICE_CALLABLE constexpr decltype(auto) type_dispatcher(
201  cudf::data_type dtype, Functor f, Ts&&... args) {
202  switch (dtype.id()) {
203  case INT8:
204  return f.template operator()<typename IdTypeMap<INT8>::type>(
205  std::forward<Ts>(args)...);
206  case INT16:
207  return f.template operator()<typename IdTypeMap<INT16>::type>(
208  std::forward<Ts>(args)...);
209  case INT32:
210  return f.template operator()<typename IdTypeMap<INT32>::type>(
211  std::forward<Ts>(args)...);
212  case INT64:
213  return f.template operator()<typename IdTypeMap<INT64>::type>(
214  std::forward<Ts>(args)...);
215  case FLOAT32:
216  return f.template operator()<typename IdTypeMap<FLOAT32>::type>(
217  std::forward<Ts>(args)...);
218  case FLOAT64:
219  return f.template operator()<typename IdTypeMap<FLOAT64>::type>(
220  std::forward<Ts>(args)...);
221  case STRING:
222  return f.template operator()<typename IdTypeMap<STRING>::type>(
223  std::forward<Ts>(args)...);
224  case TIMESTAMP_DAYS:
225  return f.template operator()<typename IdTypeMap<TIMESTAMP_DAYS>::type>(
226  std::forward<Ts>(args)...);
227  case TIMESTAMP_SECONDS:
228  return f.template operator()<typename IdTypeMap<TIMESTAMP_SECONDS>::type>(
229  std::forward<Ts>(args)...);
230  case TIMESTAMP_MILLISECONDS:
231  return f.template operator()<typename IdTypeMap<TIMESTAMP_MILLISECONDS>::type>(
232  std::forward<Ts>(args)...);
233  case TIMESTAMP_MICROSECONDS:
234  return f.template operator()<typename IdTypeMap<TIMESTAMP_MICROSECONDS>::type>(
235  std::forward<Ts>(args)...);
236  case TIMESTAMP_NANOSECONDS:
237  return f.template operator()<typename IdTypeMap<TIMESTAMP_NANOSECONDS>::type>(
238  std::forward<Ts>(args)...);
239  default: {
240 #ifndef __CUDA_ARCH__
241  CUDF_FAIL("Unsupported type_id.");
242 #else
243  release_assert(false && "Unsuported type_id.");

253  return return_type();
254 #endif
255  }
