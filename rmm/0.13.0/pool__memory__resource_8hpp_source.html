133  block block_from_stream(free_list& blocks,
134  cudaStream_t blocks_stream,
135  size_t size,
136  cudaStream_t stream) {
137  block const b = blocks.best_fit(size); // get the best fit block
138 
139  // If we found a block associated with a different stream,
140  // we have to synchronize the stream in order to use it
141  if ((blocks_stream != stream) && (b.ptr != nullptr)) {
142  cudaError_t result = cudaStreamSynchronize(blocks_stream);
143 
144  RMM_EXPECTS((result == cudaSuccess || // stream synced
145  result == cudaErrorInvalidResourceHandle), // stream deleted
146  rmm::bad_alloc, "cudaStreamSynchronize failure");
147 
148  // Now that this stream is synced, insert all other blocks into this stream's list
149  // Note: This could cause thrashing between two streams. On the other hand, it reduces
150  // fragmentation by coalescing.
151  stream_free_blocks_[stream].insert(blocks.begin(), blocks.end());
152 
153  // remove this stream from the freelist
154  stream_free_blocks_.erase(blocks_stream);
155  }
156  return b;
157  }

171  block available_larger_block(size_t size, cudaStream_t stream) {
172  // Try to find a larger block in free list for the same stream
173  auto iter = stream_free_blocks_.find(stream);
174  if (iter != stream_free_blocks_.end()) {
175  block b = block_from_stream(iter->second, stream, size, stream);
176  if (b.ptr != nullptr) return b;
177  }
178 
179  // nothing in this stream's free list, look for one on another stream
180  auto s = stream_free_blocks_.begin();
181  while (s != stream_free_blocks_.end()) {
182  if (s->first != stream) {
183  block b = block_from_stream(s->second, s->first, size, stream);
184  if (b.ptr != nullptr) return b;
185  }
186  ++s;
187  }
188 
189  // no larger blocks available on other streams, so grow the pool and create a block
190  size_t grow_size = size_to_grow(size);
191  RMM_EXPECTS(grow_size > 0, rmm::bad_alloc, "Maximum pool size exceeded");
192  return block_from_upstream(grow_size, stream);
193  }

205  void* allocate_from_block(block const& b, size_t size, cudaStream_t stream)
206  {
207  block const alloc{b.ptr, size, b.is_head};
208 
209  if (b.size > size)
210  {
211  block rest{b.ptr + size, b.size - size, false};
212  stream_free_blocks_[stream].insert(rest);
213  }
214 
215  allocated_blocks_.insert(alloc);
216  return reinterpret_cast<void*>(alloc.ptr);
217  }

226  void free_block(void *p, size_t size, cudaStream_t stream)
227  {
228  if (p == nullptr) return;
229 
230  auto const i = allocated_blocks_.find(block{static_cast<char*>(p)});
231  assert(i != allocated_blocks_.end());
232  assert(i->size == rmm::detail::align_up(size, allocation_alignment));
233 
234  stream_free_blocks_[stream].insert(*i);
235  allocated_blocks_.erase(i);
236  }

247  size_t size_to_grow(size_t size) const {
248  auto const remaining = rmm::detail::align_up(maximum_pool_size_ - pool_size(),
249  allocation_alignment);
250  auto const aligned_size = rmm::detail::align_up(size, allocation_alignment);
251  if (aligned_size <= remaining / 2) { return remaining / 2; }
252  else if (aligned_size <= remaining) { return remaining; }
253  else { return 0; }
254  };

263  block block_from_upstream(size_t size, cudaStream_t stream)
264  {
265  void* p = upstream_mr_->allocate(size, stream);
266  block b{reinterpret_cast<char*>(p), size, true};
267  upstream_blocks_.emplace_back(b);
268  current_pool_size_ += b.size;
