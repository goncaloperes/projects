46  // A block is the fixed size this resource alloates
47  static constexpr std::size_t default_block_size = 1<<20; // 1 MiB
48  // This is the number of blocks that the pool starts out with, and also the number of
49  // blocks by which the pool grows when all of its current blocks are allocated

115  // blocks are maintained in a simple list of pointers

130  void* do_allocate(std::size_t bytes, cudaStream_t stream) override {
131  if (bytes <= 0) return nullptr;
132  bytes = rmm::detail::align_up(bytes, allocation_alignment);
133  RMM_EXPECTS(bytes <= block_size_, rmm::bad_alloc, "bytes must be <= block_size");
134

148  void do_deallocate(void* p, std::size_t bytes, cudaStream_t stream) override {
149  bytes = rmm::detail::align_up(bytes, allocation_alignment);
150  assert(bytes <= block_size_);
151  stream_blocks_[stream].push_back(p);
152  }

176  void* block_from_stream(free_list& blocks,
177  cudaStream_t blocks_stream,
178  cudaStream_t stream) {
179  void* p = nullptr;
180 
181  if (!blocks.empty()) {
182  p = blocks.front();
183  blocks.pop_front();
184  }
185 
186  // If we found a block associated with a different stream,
187  // we have to synchronize the stream in order to use it
188  if ((blocks_stream != stream) && (p != nullptr)) {
189  cudaError_t result = cudaStreamSynchronize(blocks_stream);
190 
191  if (result != cudaErrorInvalidResourceHandle && // stream deleted
192  result != cudaSuccess) // stream synced
193  throw std::runtime_error{"cudaStreamSynchronize failure"};
194 
195  // insert all other blocks into this stream's list
196  // Note: This could cause thrashing between two streams. For future analysis.
197  auto blocks_for_stream = stream_blocks_[stream];
198  blocks_for_stream.splice(blocks_for_stream.end(), blocks);
199 
200  // remove this stream from the freelist
201  stream_blocks_.erase(blocks_stream);
202  }
203 
204  return p;
205  }
206 
218  void* get_block(cudaStream_t stream) {
219  // Try to find a block in the same stream
220  auto iter = stream_blocks_.find(stream);
221  if (iter != stream_blocks_.end()) {
222  void* p = block_from_stream(iter->second, stream, stream);
223  if (p != nullptr) return p;
224  }
225 
226  // nothing in this stream's free list, look for one on another stream
227  // Try to find a larger block in a different stream
228  auto s = stream_blocks_.begin();
229  while (s != stream_blocks_.end()) {
230  if (s->first != stream) {
231  void* p = block_from_stream(s->second, s->first, stream);
232  if (p != nullptr) return p;
233  }
234  ++s;
235  }
236 
237  // nothing available in other streams, get new blocks
238  // avoid searching for this stream's list again
239  free_list* plist = (iter != stream_blocks_.end()) ? &iter->second : &stream_blocks_[stream];
240 
241  new_blocks_from_upstream(stream, *plist);
242  return get_block(stream);
243  }
244 
245  //
252  void new_blocks_from_upstream(cudaStream_t stream, free_list& blocks) {
253  void* p = upstream_mr_->allocate(upstream_chunk_size_, stream);
254  upstream_blocks_.push_back(p);
255 
256  auto num_blocks = upstream_chunk_size_ / block_size_;
257 
258  auto g = [p, this](int i) { return static_cast<char*>(p) + i * block_size_; };
259  auto first = thrust::make_transform_iterator(thrust::make_counting_iterator(std::size_t{0}), g);
260  blocks.insert(blocks.cend(), first, first + num_blocks);
261  }
262 
267  void release()
268  {
269  for (auto p : upstream_blocks_)
270  upstream_mr_->deallocate(p, upstream_chunk_size_);
271  upstream_blocks_.clear();
272  stream_blocks_.clear();
273  }
274 
275  Upstream* upstream_mr_; // The resource from which to allocate new blocks
276 
277  std::size_t const block_size_; // size of blocks this MR allocates
278  std::size_t const upstream_chunk_size_; // size of chunks allocated from heap MR
279 
280  // stream free lists: map of [stream_id, free_list] pairs
281  // stream stream_id must be synced before allocating from this list
282  std::map<cudaStream_t, free_list> stream_blocks_;
283 
284  // blocks allocated from heap: so they can be easily freed
285  std::vector<void*> upstream_blocks_;
286 };
